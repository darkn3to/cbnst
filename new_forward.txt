#include <stdio.h>

void print_table(double x[], double y[][10], int n) {
    printf("\nDivided Difference Table:\n");
    for (int i = 0; i < n; i++) {
        printf("%.2lf", x[i]);
        for (int j = 0; j < n - i; j++) {
            printf("\t%.6lf", y[i][j]);
        }
        printf("\n");
    }
}

void newton_forward(double x[], double y[][10], int n, double val) {
    double res = y[0][0];  // Result of interpolation
    double u = (val - x[0]) / (x[1] - x[0]);  // u calculation
    double temp_u = u;  // Temporary variable for u terms
    double fact = 1;  // Factorial variable

    // Newton's forward interpolation formula
    for (int i = 1; i < n; i++) {
        res += (temp_u * y[0][i]) / fact;
        temp_u *= (u - i);
        fact *= (i + 1);
    }
    
    printf("The interpolated value at %.2lf is %.6lf\n", val, res);
}

int main() {
    int n;
    double x[10], y[10][10], val;

    printf("Enter the number of data points: ");
    scanf("%d", &n);

    printf("Enter the x values:\n");
    for (int i = 0; i < n; i++) {
        scanf("%lf", &x[i]);
    }

    printf("Enter the y values:\n");
    for (int i = 0; i < n; i++) {
        scanf("%lf", &y[i][0]);
    }

    // Divided difference table calculation
    for (int j = 1; j < n; j++) {
        for (int i = 0; i < n - j; i++) {
            y[i][j] = y[i + 1][j - 1] - y[i][j - 1];
        }
    }

    print_table(x, y, n);

    printf("Enter the value to interpolate: ");
    scanf("%lf", &val);

    newton_forward(x, y, n, val);

    return 0;
}
